1. Get the application URL by running these commands:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ include "smc-trading-agent.namespace" . }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "smc-trading-agent.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ include "smc-trading-agent.namespace" . }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get --namespace {{ include "smc-trading-agent.namespace" . }} svc -w {{ include "smc-trading-agent.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ include "smc-trading-agent.namespace" . }} {{ include "smc-trading-agent.fullname" . }} --template "{{ "{{" }} range (index .status.loadBalancer.ingress 0) {{ "}}" }}{{ "{{" }}.{{ "}}" }}{{ "{{" }} end {{ "}}" }}")
  echo http://$SERVICE_IP:{{ .Values.service.ports.http.port }}
{{- else if contains "ClusterIP" .Values.service.type }}
  export POD_NAME=$(kubectl get pods --namespace {{ include "smc-trading-agent.namespace" . }} -l "app.kubernetes.io/name={{ include "smc-trading-agent.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  export CONTAINER_PORT=$(kubectl get pod --namespace {{ include "smc-trading-agent.namespace" . }} $POD_NAME -o jsonpath="{.spec.containers[0].ports[0].containerPort}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl --namespace {{ include "smc-trading-agent.namespace" . }} port-forward $POD_NAME 8080:$CONTAINER_PORT
{{- end }}
